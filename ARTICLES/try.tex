    \If{$\text{is\_gold\_market\_open\_today}$}
        \State\text{compute\_max\_profit\_and\_optimal\_type}

    \EndIf
    \Comment{update\_state}

        \State$\text{end\_state}[\text{current\_day} + 1] \gets [\text{computed\_profit}, \ldots]$
\EndFor
\end{algorithmic}
